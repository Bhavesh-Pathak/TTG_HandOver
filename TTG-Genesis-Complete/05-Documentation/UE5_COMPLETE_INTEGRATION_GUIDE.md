│   │   │   ├── YourWorldNameQuestSystem.cpp

│   │   └── YourWorldName.Build.cs  # Build configuration
│   ├── YourWorldName.Target.cs     # Game target
│   └── YourWorldNameEditor.Target.cs # Editor target
├── Content/                        # Game content
│   └── TTGGenesis/
│       ├── Blueprints/             # Blueprint data

│       ├── Data/                   # NPC/Quest data
